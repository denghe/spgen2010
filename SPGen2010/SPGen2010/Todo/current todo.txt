do
{
    var hi = reader.GetSchemaTable();
    var dt = new DbTable();
    for (int i = 0; i < hi.Rows.Count; i++)
        dt.NewColumn((string)hi.Rows[i]["ColumnName"]
            , (Type)hi.Rows[i]["DataType"]
            , (bool)hi.Rows[i]["AllowDBNull"]);
    ds.Tables.Add(dt);
    if (reader.HasRows)
        while (reader.Read())
        {
            var buffer = new object[dt.Columns.Count];
            reader.GetValues(buffer);
            dt.NewRow(buffer);
        }
} while (reader.NextResult());
