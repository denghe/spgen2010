using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Runtime.InteropServices;
using System.Data.SqlClient;
using System.Data;
using Microsoft.SqlServer.Server;
using System.Diagnostics;

namespace ConsoleApplication1
{
    class Program
    {
        static void Main(string[] args)
        {
            var sw = new Stopwatch();

            sw.Start();
            for (int i = 0; i < 100000000; i++)
            {
                var foo = Foo.Instance;
            }
            sw.Stop();
            Console.WriteLine(sw.ElapsedMilliseconds);
            sw.Restart();
            for (int i = 0; i < 100000000; i++)
            {
                var bar = Bar.GetInstance();
            }
            sw.Stop();
            Console.WriteLine(sw.ElapsedMilliseconds);
        }
    }

    public class Foo
    {
        private Foo() { }
        private static Foo _instance = null;
        public static Foo Instance
        {
            get
            {
                if (_instance == null) _instance = new Foo();
                return _instance;
            }
        }
    }

    public class Bar
    {
        private Bar() { }
        private static Bar _instance = null;
        public static Func<Bar> GetInstance = () =>
        {
            _instance = new Bar();
            GetInstance = () => { return _instance; };
            return _instance;
        };
    }
}
